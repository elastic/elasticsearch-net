[[dismax-query-usage]]
=== Dismax Query Usage

==== Fluent DSL example

[source,csharp]
----
q
.DisMax(c => c
    .Name("named_query")
    .Boost(1.1)
    .TieBreaker(0.11)
    .Queries(
        qq => qq.MatchAll(m => m.Name("query1")),
        qq => qq.MatchAll(m => m.Name("query2"))
    )
)
----

==== Object Initializer syntax example

[source,csharp]
----
new DisMaxQuery()
{
    Name = "named_query",
    Boost = 1.1,
    TieBreaker = 0.11,
    Queries = new QueryContainer[]
    {
        new MatchAllQuery() { Name = "query1" },
        new MatchAllQuery() { Name = "query2" },
    }
}
----
